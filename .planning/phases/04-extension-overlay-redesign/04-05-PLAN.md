const fs = require('fs');
const t = fs.readFileSync('packages/tokens/tokens.css', 'utf8');
const o = fs.readFileSync('apps/extension/src/overlay.ts', 'utf8');

// Extract animation: rules outside @media blocks
function checkFile(name, content) {
  const lines = content.split('\n');
  let inMedia = false; let depth = 0;
  const animUses = [];
  const transitionUses = [];
  for (let i = 0; i < lines.length; i++) {
    const l = lines[i];
    if (l.includes('@media') && l.includes('prefers-reduced-motion')) inMedia = true;
    if (inMedia) { for (const c of l) { if (c === '{') depth++; if (c === '}') { depth--; if (depth === 0) inMedia = false; } } continue; }
    if (/animation:\s*[a-zA-Z]/.test(l) && !l.includes('none')) animUses.push((i+1) + ': ' + l.trim());
    if (/transition:\s*(width|transform|all|background|opacity)/.test(l) && !l.includes('none')) transitionUses.push((i+1) + ': ' + l.trim());
  }
  console.log(name, 'animations outside reduced-motion:', animUses.length);
  console.log(name, 'transitions outside reduced-motion:', transitionUses.length);
}
checkFile('tokens.css', t);
checkFile('overlay.ts', o);
console.log('Audit complete (informational — non-blocking)');
